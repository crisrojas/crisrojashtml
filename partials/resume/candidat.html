class Candidat : NSObject {

var name: String!

var age: Int!

var photo: UIImage!

var skills: [String]!

var hardSkills: [String]!

var softSkills: [String]!

var langues: [String]!

var hobbies: [String]!

init(name: String,

photo: UIImage,

skills: [String],

hardSkills: [String],

softSkills: [String],

hobbies: [String:String],

langues: [String:String]) {

self.name = name

self.photo = photo

self.skills = skills

self.hardSkills = hardSkills

self.softSkills = softSkills

self.hobbies = hobbies

self.langues = langues

}

}
